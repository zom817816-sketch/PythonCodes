```python
def select_sort(items, comp=lambda x, y: x < y): 
	items = items[:] 
	for i in range(len(items)-1): 
		min_index = i 
		for j in range(i+1, len(items)): 
			if comp(items[j], items[min_index]): 
				min_index = j 
		items[i], items[min_index] = items[min_index], items[i] 
	return items
```
- 冒泡排序升级（搅拌排序）
```python
def bubble_sort(items, comp=lambda x, y: x < y): 
	items = items[:] 
	for i in range(len(items)-1): 
		swapped = False 
		for j in range(len(items)-1-i): 
			if comp(items[j+1], items[j]): 
				items[j+1], items[j] = items[j], items[j+1] 
				swapped = True 
		if swapped: 
			swapped = False 
			for j in range(len(items)-2-i, i, -1): 
				if comp(items[j], items[j-1]): 
					items[j-1], items[j] = items[j], items[j-1] 
		if not swapped: 
			break 
	return items
```
- 归并排序
```python
def merge(items1, items2, comp=lambda x, y: x < y): 
	index1, index2 = 0, 0   
	merged_items = [] 
	while index1 < len(items1) and index2 < len(items2): 
		if comp(items1[index1], items2[index2]): 
			merged_items.append(items1[index1]) 
			index1 += 1 
		else: 
			merged_items.append(items2[index2]) 
			index2 += 1 
	merged_items += items1[index1:] 
	merged_items += items2[index2:] 
	return merged_items 
def merge_sort(items, comp=lambda x, y: x < y): 
	items = list(items[:]) 
	return _merge_sort(items, comp) 
def _merge_sort(items, comp): 
	if len(items) < 2: 
		return items
	mid = len(items) // 2 
	left = _merge_sort(items[:mid], comp) 
	right = _merge_sort(items[mid:], comp) 
	return merge(left, right, comp) 
``` 
- 快速排序

```python
def quick_sort(items, comp=lambda x, y: x < y): 
	items = list(items[:]) 
	_quick_sort(items, 0, len(items)-1, comp) 
	return items 
def _quick_sort(items, start, end, comp): 
	if start < end: 
		pos = _partition(items, start, end, comp) 
		_quick_sort(items, start, pos-1, comp) 
		_quick_sort(items, pos+1, end, comp) 
def _partition(items, start, end, comp): 
	pivot = items[end] 
	i = start - 1 
	for j in range(start, end): 
		if comp(items[j], pivot): 
			items[j], items[i] = items[i], items[j] 
			i += 1 
	items[i+1], items[end] = items[end], items[i+1] 
	return i+1 
```
- 基数排序(非负整数)
```Python 
def radix_sort(items): 
	# 处理空数组
	if not items: 
		return items 
	# 找到数组中最大数，确定最大位数
	max_num = max(items) 
	max_digit = len(str(max_num)) 
	# 初始化10个数位桶（0-9）+ 1个主桶
	main_bucket = list(items[:]) 
	digit_buckets = [[] for _ in range(10)] 
	# 按每一位数字排序
	for digit in range(max_digit): 
		# 把主桶元素分配到对应数位桶
		for num in main_bucket: 
			digit_buckets[num // (10 ** digit) % 10].append(num) 
		main_bucket = []
		# 把10个数位桶按顺序合并，重新赋值给主桶
		for bucket in digit_buckets: 
			main_bucket.extend(bucket) 
			bucket.clear() # 清空数位桶，供下一轮使用
	
	return main_bucket
```
